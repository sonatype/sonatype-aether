33      /**
34       * Creates a version range from the specified range specification.
35       * 
36       * @param range The range specification to parse, must not be {@code null}.
37       * @throws InvalidVersionSpecificationException If the range could not be parsed.
38       */
39      public TestVersionRange( String range )
40          throws InvalidVersionSpecificationException
41      {
42          String process = range;
43  
44          if ( range.startsWith( "[" ) )
45          {
46              lowerBoundInclusive = true;
47          }
48          else if ( range.startsWith( "(" ) )
49          {
50              lowerBoundInclusive = false;
51          }
52          else
53          {
54              throw new InvalidVersionSpecificationException( range, "Invalid version range " + range
55                  + ", a range must start with either [ or (" );
56          }
57  
58          if ( range.endsWith( "]" ) )
59          {
60              upperBoundInclusive = true;
61          }
62          else if ( range.endsWith( ")" ) )
63          {
64              upperBoundInclusive = false;
65          }
66          else
67          {
68              throw new InvalidVersionSpecificationException( range, "Invalid version range " + range
69                  + ", a range must end with either [ or (" );
70          }
71  
72          process = process.substring( 1, process.length() - 1 );
73  
74          int index = process.indexOf( "," );
75  
76          if ( index < 0 )
77          {
78              if ( !lowerBoundInclusive || !upperBoundInclusive )
79              {
80                  throw new InvalidVersionSpecificationException( range, "Invalid version range " + range
81                      + ", single version must be surrounded by []" );
82              }
83  
84              lowerBound = upperBound = new StubVersion( process.trim() );
85          }
86          else
87          {
88              String parsedLowerBound = process.substring( 0, index ).trim();
89              String parsedUpperBound = process.substring( index + 1 ).trim();
90  
91              // more than two bounds, e.g. (1,2,3)
92              if ( parsedUpperBound.contains( "," ) )
93              {
94                  throw new InvalidVersionSpecificationException( range, "Invalid version range " + range
95                      + ", bounds may not contain additional ','" );
96              }
97  
98              lowerBound =  parsedLowerBound.length() > 0 ? new StubVersion( parsedLowerBound ) : null;
99              upperBound = parsedUpperBound.length() > 0 ? new StubVersion( parsedUpperBound ) : null;
100 
101             if ( upperBound != null && lowerBound != null )
102             {
103                 if ( upperBound.compareTo( lowerBound ) < 0 )
104                 {
105                     throw new InvalidVersionSpecificationException( range, "Invalid version range " + range
106                         + ", lower bound must not be greater than upper bound" );
107                 }
108             }
109         }
110     }

146     public boolean containsVersion( Version version )
147     {
148         boolean snapshot = isSnapshot( version );
149 
150         if ( lowerBound != null )
151         {
152             int comparison = lowerBound.compareTo( version );
153 
154             if ( snapshot && comparison == 0 )
155             {
156                 return true;
157             }
158 
159             if ( comparison == 0 && !lowerBoundInclusive )
160             {
161                 return false;
162             }
163             if ( comparison > 0 )
164             {
165                 return false;
166             }
167         }
168 
169         if ( upperBound != null )
170         {
171             int comparison = upperBound.compareTo( version );
172 
173             if ( snapshot && comparison == 0 )
174             {
175                 return true;
176             }
177 
178             if ( comparison == 0 && !upperBoundInclusive )
179             {
180                 return false;
181             }
182             if ( comparison < 0 )
183             {
184                 return false;
185             }
186         }
187 
188         if ( lowerBound != null || upperBound != null )
189         {
190             return !snapshot;
191         }
192 
193         return true;
194     }

196     private boolean isSnapshot( Version version )
197     {
198         return version != null && version.toString().endsWith( "SNAPSHOT" );
