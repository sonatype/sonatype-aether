17  /**
18   * A request to register an artifact within the local repository. Certain local repository implementations can refuse to
19   * serve physically present artifacts if those haven't been previously registered to them.
20   * 
21   * @author Benjamin Bentmann
22   * @see LocalRepositoryManager#add(RepositorySystemSession, LocalArtifactRegistration)
23   */
24  public class LocalArtifactRegistration
25  {
26  
27      private Artifact artifact;
28  
29      private RemoteRepository repository;
30  
31      private Collection<String> contexts = Collections.emptyList();
32  
33      /**
34       * Creates an uninitialized registration.
35       */
36      public LocalArtifactRegistration()
37      {
38          // enables default constructor
39      }
40  
41      /**
42       * Creates a registration request for the specified (locally installed) artifact.
43       * 
44       * @param artifact The artifact to register, may be {@code null}.
45       */
46      public LocalArtifactRegistration( Artifact artifact )
47      {
48          setArtifact( artifact );
49      }
50  
51      /**
52       * Creates a registration request for the specified artifact.
53       * 
54       * @param artifact The artifact to register, may be {@code null}.
55       * @param repository The remote repository from which the artifact was resolved or {@code null} if the artifact was
56       *            locally installed.
57       * @param contexts The resolution contexts, may be {@code null}.
58       */
59      public LocalArtifactRegistration( Artifact artifact, RemoteRepository repository, Collection<String> contexts )
60      {
61          setArtifact( artifact );
62          setRepository( repository );
63          setContexts( contexts );
64      }
65  
66      /**
67       * Gets the artifact to register.
68       * 
69       * @return The artifact or {@code null} if not set.
70       */
71      public Artifact getArtifact()

73          return artifact;
74      }
75  
76      /**
77       * Sets the artifact to register.
78       * 
79       * @param artifact The artifact, may be {@code null}.
80       * @return This request for chaining, never {@code null}.
81       */
82      public LocalArtifactRegistration setArtifact( Artifact artifact )
83      {
84          this.artifact = artifact;
85          return this;
86      }
87  
88      /**
89       * Gets the remote repository from which the artifact was resolved.
90       * 
91       * @return The remote repository or {@code null} if the artifact was locally installed.
92       */
93      public RemoteRepository getRepository()
94      {
95          return repository;
96      }
97  
98      /**
99       * Sets the remote repository from which the artifact was resolved.
100      * 
101      * @param repository The remote repository or {@code null} if the artifact was locally installed.
102      * @return This request for chaining, never {@code null}.
103      */
104     public LocalArtifactRegistration setRepository( RemoteRepository repository )
105     {
106         this.repository = repository;
107         return this;
108     }
109 
110     /**
111      * Gets the resolution contexts in which the artifact is available.
112      * 
113      * @return The resolution contexts in which the artifact is available, never {@code null}.
114      */
115     public Collection<String> getContexts()
116     {
117         return contexts;
118     }
119 
120     /**
121      * Sets the resolution contexts in which the artifact is available.
122      * 
123      * @param contexts The resolution contexts, may be {@code null}.
124      * @return This request for chaining, never {@code null}.
125      */
126     public LocalArtifactRegistration setContexts( Collection<String> contexts )
127     {
128         if ( contexts != null )
129         {
130             this.contexts = contexts;
131         }
132         else
133         {
134             this.contexts = Collections.emptyList();
135         }

137     }
