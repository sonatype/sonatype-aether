132     private static class FlakyHandler
133         extends AbstractHandler
134     {
135 
136         private static final Pattern RANGE = Pattern.compile( "bytes=([0-9]+)-" );
137 
138         private final int requiredRequests;
139 
140         private final Map<String, Integer> madeRequests;
141 
142         private final int totalSize;
143 
144         private final int chunkSize;
145 
146         public FlakyHandler( int requiredRequests )
147         {
148             this.requiredRequests = requiredRequests;
149             madeRequests = new ConcurrentHashMap<String, Integer>();
150 
151             totalSize = 1024 * 128;
152             chunkSize = ( requiredRequests > 1 ) ? totalSize / ( requiredRequests - 1 ) - 1 : totalSize;
153         }
154 
155         public void handle( String target, Request baseRequest, HttpServletRequest request, HttpServletResponse response )
156             throws IOException
157         {
158             Integer attempts = madeRequests.get( target );
159             attempts = ( attempts == null ) ? Integer.valueOf( 1 ) : Integer.valueOf( attempts.intValue() + 1 );
160             madeRequests.put( target, attempts );
161 
162             if ( attempts.intValue() > requiredRequests )
163             {
164                 response.setStatus( HttpURLConnection.HTTP_BAD_REQUEST );
165                 response.flushBuffer();
166                 return;
167             }
168 
169             int lb = 0, ub = totalSize - 1;
170 
171             String range = request.getHeader( "Range" );
172             if ( range != null && range.matches( RANGE.pattern() ) )
173             {
174                 Matcher m = RANGE.matcher( range );
175                 m.matches();
176                 lb = Integer.parseInt( m.group( 1 ) );
177             }
178 
179             response.setStatus( ( lb > 0 ) ? HttpURLConnection.HTTP_PARTIAL : HttpURLConnection.HTTP_OK );
180             response.setContentLength( totalSize - lb );
181             response.setContentType( "Content-type: text/plain; charset=UTF-8" );
182             if ( lb > 0 )
183             {
184                 response.setHeader( "Content-Range", "bytes " + lb + "-" + ub + "/" + totalSize );
185             }
186             response.flushBuffer();
187 
188             OutputStream out = response.getOutputStream();
189 
190             for ( int i = lb, j = 0; i <= ub; i++, j++ )
191             {
192                 if ( j >= chunkSize )
193                 {
194                     out.flush();
195 
196                     throw new IOException( "oups, we're dead" );
197                 }
198 
199                 out.write( CONTENT_PATTERN[i % CONTENT_PATTERN.length] );
200             }
201 
202             out.close();
203         }
204 
205     }
