11  import java.util.Collection;
12  import java.util.Collections;
13  import java.util.LinkedHashSet;
14  import java.util.List;

16  import org.sonatype.aether.graph.DependencyFilter;

19  /**
20   * A dependency filter that combines zero or more other filters using a logical {@code OR}.
21   * 
22   * @author Benjamin Bentmann
23   */
24  public class OrDependencyFilter
25      implements DependencyFilter
26  {
27  
28      private final Collection<DependencyFilter> filters = new LinkedHashSet<DependencyFilter>();
29  
30      /**
31       * Creates a new filter from the specified filters.
32       * 
33       * @param filters The filters to combine, may be {@code null}.
34       */
35      public OrDependencyFilter( DependencyFilter... filters )
36      {
37          if ( filters != null )
38          {
39              Collections.addAll( this.filters, filters );
40          }
41      }
42  
43      /**
44       * Creates a new filter from the specified filters.
45       * 
46       * @param filters The filters to combine, may be {@code null}.
47       */
48      public OrDependencyFilter( Collection<DependencyFilter> filters )
49      {
50          if ( filters != null )
51          {
52              this.filters.addAll( filters );
53          }
54      }
55  
56      /**
57       * Creates a new filter from the specified filters.
58       * 
59       * @param filter1 The first filter to combine, may be {@code null}.
60       * @param filter2 The first filter to combine, may be {@code null}.
61       * @return The combined filter or {@code null} if both filter were {@code null}.
62       */
63      public static DependencyFilter newInstance( DependencyFilter filter1, DependencyFilter filter2 )
64      {
65          if ( filter1 == null )
66          {
67              return filter2;
68          }
69          else if ( filter2 == null )
70          {
71              return filter1;
72          }
73          return new OrDependencyFilter( filter1, filter2 );
74      }
75  
76      public boolean accept( DependencyNode node, List<DependencyNode> parents )
77      {
78          for ( DependencyFilter filter : filters )
79          {
80              if ( filter.accept( node, parents ) )
81              {
82                  return true;
83              }
84          }
85          return false;
86      }
87  
88      @Override
89      public boolean equals( Object obj )
90      {
91          if ( this == obj )
92          {
93              return true;
94          }
95  
96          if ( obj == null || !getClass().equals( obj.getClass() ) )
97          {
98              return false;
99          }
100 
101         OrDependencyFilter that = (OrDependencyFilter) obj;
102 
103         return this.filters.equals( that.filters );
104     }
105 
106     @Override
107     public int hashCode()
108     {
109         int hash = getClass().hashCode();
110         hash = hash * 31 + filters.hashCode();
111         return hash;
112     }
