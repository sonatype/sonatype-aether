19  import org.sonatype.aether.impl.LocalRepositoryProvider;

30   * @author Benjamin Bentmann
31   */
32  @Component( role = LocalRepositoryProvider.class )
33  public class DefaultLocalRepositoryProvider
34      implements LocalRepositoryProvider, Service
35  {
36  
37      @Requirement
38      private Logger logger = NullLogger.INSTANCE;
39  
40      @Requirement( role = LocalRepositoryManagerFactory.class )
41      private List<LocalRepositoryManagerFactory> managerFactories = new ArrayList<LocalRepositoryManagerFactory>();
42  
43      private static final Comparator<LocalRepositoryManagerFactory> COMPARATOR =
44          new Comparator<LocalRepositoryManagerFactory>()
45          {
46  
47              public int compare( LocalRepositoryManagerFactory o1, LocalRepositoryManagerFactory o2 )
48              {
49                  return o2.getPriority() - o1.getPriority();
50              }
51  
52          };
53  
54      public DefaultLocalRepositoryProvider()
55      {
56          // enables default constructor
57      }
58  
59      public void initService( ServiceLocator locator )
60      {
61          setLogger( locator.getService( Logger.class ) );
62          setLocalRepositoryManagerFactories( locator.getServices( LocalRepositoryManagerFactory.class ) );
63      }
64  
65      public DefaultLocalRepositoryProvider setLogger( Logger logger )
66      {
67          this.logger = ( logger != null ) ? logger : NullLogger.INSTANCE;
68          return this;
69      }
70  
71      public DefaultLocalRepositoryProvider setLocalRepositoryManagerFactories( List<LocalRepositoryManagerFactory> factories )
72      {
73          if ( factories == null )
74          {
75              managerFactories = new ArrayList<LocalRepositoryManagerFactory>( 2 );
76          }
77          else
78          {
79              managerFactories = factories;
80          }
81          return this;
82      }
83  
84      public DefaultLocalRepositoryProvider addLocalRepositoryManagerFactory( LocalRepositoryManagerFactory factory )
85      {
86          if ( factory == null )
87          {
88              throw new IllegalArgumentException( "Local repository manager factory has not been specified." );
89          }
90          managerFactories.add( factory );
91          return this;
92      }
93  
94      public LocalRepositoryManager newLocalRepositoryManager( LocalRepository localRepository )
95          throws NoLocalRepositoryManagerException
96      {
97          List<LocalRepositoryManagerFactory> factories = new ArrayList<LocalRepositoryManagerFactory>( managerFactories );
98          Collections.sort( factories, COMPARATOR );
99  
100         for ( LocalRepositoryManagerFactory factory : factories )
101         {
102             try
103             {
104                 LocalRepositoryManager manager = factory.newInstance( localRepository );
105 
106                 if ( logger.isDebugEnabled() )
107                 {
108                     StringBuilder buffer = new StringBuilder( 256 );
109                     buffer.append( "Using manager " ).append( manager.getClass().getSimpleName() );
110                     buffer.append( " with priority " ).append( factory.getPriority() );
111                     buffer.append( " for " ).append( localRepository.getBasedir() );
112 
113                     logger.debug( buffer.toString() );
114                 }
115 
116                 return manager;
117             }
118             catch ( NoLocalRepositoryManagerException e )
119             {
120                 // continue and try next factory
121             }
122         }
123 
124         StringBuilder buffer = new StringBuilder( 256 );
125         buffer.append( "No manager available for local repository " );
126         buffer.append( localRepository.getBasedir() );
127         buffer.append( " of type " ).append( localRepository.getContentType() );
128         buffer.append( " using the available factories " );
129         for ( ListIterator<LocalRepositoryManagerFactory> it = factories.listIterator(); it.hasNext(); )
130         {
131             LocalRepositoryManagerFactory factory = it.next();
132             buffer.append( factory.getClass().getSimpleName() );
133             if ( it.hasNext() )
134             {
135                 buffer.append( ", " );
136             }
137         }
138 
139         throw new NoLocalRepositoryManagerException( localRepository, buffer.toString() );
140     }
