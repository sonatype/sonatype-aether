11  import org.codehaus.plexus.component.annotations.Component;
12  import org.codehaus.plexus.component.annotations.Requirement;
13  import org.sonatype.aether.repository.LocalRepository;
14  import org.sonatype.aether.repository.LocalRepositoryManager;
15  import org.sonatype.aether.repository.NoLocalRepositoryManagerException;
16  import org.sonatype.aether.spi.localrepo.LocalRepositoryManagerFactory;
17  import org.sonatype.aether.spi.locator.Service;
18  import org.sonatype.aether.spi.locator.ServiceLocator;
19  import org.sonatype.aether.spi.log.Logger;
20  import org.sonatype.aether.spi.log.NullLogger;

22  /**
23   * Creates {@link EnhancedLocalRepositoryManager}s for repository types {@code "default"} or {@code "" (automatic)}.
24   * 
25   * @author Benjamin Hanzelmann
26   */
27  @Component( role = LocalRepositoryManagerFactory.class, hint = "enhanced" )
28  public class EnhancedLocalRepositoryManagerFactory
29      implements LocalRepositoryManagerFactory, Service
30  {
31  
32      @Requirement
33      private Logger logger = NullLogger.INSTANCE;
34  
35      public LocalRepositoryManager newInstance( LocalRepository repository )
36          throws NoLocalRepositoryManagerException
37      {
38          if ( "".equals( repository.getContentType() ) || "default".equals( repository.getContentType() ) )
39          {
40              return new EnhancedLocalRepositoryManager( repository.getBasedir() ).setLogger( logger );
41          }
42          else
43          {
44              throw new NoLocalRepositoryManagerException( repository );
45          }
46      }
47  
48      public void initService( ServiceLocator locator )
49      {
50          setLogger( locator.getService( Logger.class ) );
51      }
52  
53      public EnhancedLocalRepositoryManagerFactory setLogger( Logger logger )
54      {
55          this.logger = ( logger != null ) ? logger : NullLogger.INSTANCE;
56          return this;
57      }
58  
59      public int getPriority()
60      {
61          return 10;
62      }
63  
64  }
