196     public List<ArtifactResult> resolveArtifacts( RepositorySystemSession session,
197                                                   Collection<? extends ArtifactRequest> requests )
198         throws ArtifactResolutionException
199     {
200         SyncContext syncContext = syncContextFactory.newInstance( session, false );
201 
202         try
203         {
204             Collection<Artifact> artifacts = new ArrayList<Artifact>( requests.size() );
205             for ( ArtifactRequest request : requests )
206             {
207                 if ( request.getArtifact().getProperty( ArtifactProperties.LOCAL_PATH, null ) != null )
208                 {
209                     continue;
210                 }
211                 artifacts.add( request.getArtifact() );
212             }
213 
214             syncContext.acquire( artifacts, null );
215 
216             return resolve( session, requests );
217         }
218         finally
219         {
220             syncContext.release();
221         }
222     }

224     private List<ArtifactResult> resolve( RepositorySystemSession session,
225                                           Collection<? extends ArtifactRequest> requests )
226         throws ArtifactResolutionException
227     {
228         List<ArtifactResult> results = new ArrayList<ArtifactResult>( requests.size() );
229         boolean failures = false;
230 
231         LocalRepositoryManager lrm = session.getLocalRepositoryManager();
232         WorkspaceReader workspace = session.getWorkspaceReader();
233 
234         List<ResolutionGroup> groups = new ArrayList<ResolutionGroup>();
235 
236         for ( ArtifactRequest request : requests )
237         {
238             RequestTrace trace = DefaultRequestTrace.newChild( request.getTrace(), request );
239 
240             ArtifactResult result = new ArtifactResult( request );
241             results.add( result );
242 
243             Artifact artifact = request.getArtifact();
244             List<RemoteRepository> repos = request.getRepositories();
245 
246             artifactResolving( session, trace, artifact );
247 
248             String localPath = artifact.getProperty( ArtifactProperties.LOCAL_PATH, null );
249             if ( localPath != null )
250             {
251                 // unhosted artifact, just validate file
252                 File file = new File( localPath );
253                 if ( !file.isFile() )
254                 {
255                     failures = true;
256                     result.addException( new ArtifactNotFoundException( artifact, null ) );
257                 }
258                 else
259                 {
260                     artifact = artifact.setFile( file );
261                     result.setArtifact( artifact );
262                     artifactResolved( session, trace, artifact, null, result.getExceptions() );
263                 }
264                 continue;
265             }
266 
267             VersionResult versionResult;
268             try
269             {
270                 VersionRequest versionRequest = new VersionRequest( artifact, repos, request.getRequestContext() );
271                 versionRequest.setTrace( trace );
272                 versionResult = versionResolver.resolveVersion( session, versionRequest );
273             }
274             catch ( VersionResolutionException e )
275             {
276                 result.addException( e );
277                 continue;
278             }
279 
280             artifact = artifact.setVersion( versionResult.getVersion() );
281 
282             if ( versionResult.getRepository() != null )
283             {
284                 if ( versionResult.getRepository() instanceof RemoteRepository )
285                 {
286                     repos = Collections.singletonList( (RemoteRepository) versionResult.getRepository() );
287                 }
288                 else
289                 {
290                     repos = Collections.emptyList();
291                 }
292             }
293 
294             if ( workspace != null )
295             {
296                 File file = workspace.findArtifact( artifact );
297                 if ( file != null )
298                 {
299                     artifact = artifact.setFile( file );
300                     result.setArtifact( artifact );
301                     result.setRepository( workspace.getRepository() );
302                     artifactResolved( session, trace, artifact, result.getRepository(), null );
303                     continue;
304                 }
305             }
306 
307             LocalArtifactResult local =
308                 lrm.find( session, new LocalArtifactRequest( artifact, repos, request.getRequestContext() ) );
309             if ( isLocallyInstalled( local, versionResult ) )
310             {
311                 if ( local.getRepository() != null )
312                 {
313                     result.setRepository( local.getRepository() );
314                 }
315                 else
316                 {
317                     result.setRepository( lrm.getRepository() );
318                 }
319                 try
320                 {
321                     artifact = artifact.setFile( getFile( session, artifact, local.getFile() ) );
322                     result.setArtifact( artifact );
323                     artifactResolved( session, trace, artifact, result.getRepository(), null );
324                 }
325                 catch ( ArtifactTransferException e )
326                 {
327                     result.addException( e );
328                 }
329                 if ( !local.isAvailable() )
330                 {
331                     /*
332                      * NOTE: Interop with simple local repository: An artifact installed by a simple local repo manager
333                      * will not show up in the repository tracking file of the enhanced local repository. If however the
334                      * maven-metadata-local.xml tells us the artifact was installed locally, we sync the repository
335                      * tracking file.
336                      */
337                     lrm.add( session, new LocalArtifactRegistration( artifact ) );
338                 }
339                 continue;
340             }
341             else if ( local.getFile() != null )
342             {
343                 logger.debug( "Verifying availability of " + local.getFile() + " from " + repos );
344             }
345 
346             if ( session.isOffline() )
347             {
348                 Exception exception =
349                     new ArtifactNotFoundException( artifact, null, "The repository system is offline but the artifact "
350                         + artifact + " is not available in the local repository." );
351                 result.addException( exception );
352                 artifactResolved( session, trace, artifact, null, result.getExceptions() );
353                 continue;
354             }
355 
356             AtomicBoolean resolved = new AtomicBoolean( false );
357             Iterator<ResolutionGroup> groupIt = groups.iterator();
358             for ( RemoteRepository repo : repos )
359             {
360                 if ( !repo.getPolicy( artifact.isSnapshot() ).isEnabled() )
361                 {
362                     continue;
363                 }
364                 ResolutionGroup group = null;
365                 while ( groupIt.hasNext() )
366                 {
367                     ResolutionGroup t = groupIt.next();
368                     if ( t.matches( repo ) )
369                     {
370                         group = t;
371                         break;
372                     }
373                 }
374                 if ( group == null )
375                 {
376                     group = new ResolutionGroup( repo );
377                     groups.add( group );
378                     groupIt = Collections.<ResolutionGroup> emptyList().iterator();
379                 }
380                 group.items.add( new ResolutionItem( trace, artifact, resolved, result, local, repo ) );
381             }
382         }
383 
384         for ( ResolutionGroup group : groups )
385         {
386             List<ArtifactDownload> downloads = new ArrayList<ArtifactDownload>();
387             for ( ResolutionItem item : group.items )
388             {
389                 Artifact artifact = item.artifact;
390 
391                 if ( item.resolved.get() )
392                 {
393                     // resolved in previous resolution group
394                     continue;
395                 }
396 
397                 ArtifactDownload download = new ArtifactDownload();
398                 download.setArtifact( artifact );
399                 download.setRequestContext( item.request.getRequestContext() );
400                 download.setTrace( item.trace );
401                 if ( item.local.getFile() != null )
402                 {
403                     download.setFile( item.local.getFile() );
404                     download.setExistenceCheck( true );
405                 }
406                 else
407                 {
408                     String path =
409                         lrm.getPathForRemoteArtifact( artifact, group.repository, item.request.getRequestContext() );
410                     download.setFile( new File( lrm.getRepository().getBasedir(), path ) );
411                 }
412 
413                 boolean snapshot = artifact.isSnapshot();
414                 RepositoryPolicy policy =
415                     remoteRepositoryManager.getPolicy( session, group.repository, !snapshot, snapshot );
416 
417                 if ( session.isNotFoundCachingEnabled() || session.isTransferErrorCachingEnabled() )
418                 {
419                     UpdateCheck<Artifact, ArtifactTransferException> check =
420                         new UpdateCheck<Artifact, ArtifactTransferException>();
421                     check.setItem( artifact );
422                     check.setFile( download.getFile() );
423                     check.setFileValid( !download.isExistenceCheck() );
424                     check.setRepository( group.repository );
425                     check.setPolicy( policy.getUpdatePolicy() );
426                     item.updateCheck = check;
427                     updateCheckManager.checkArtifact( session, check );
428                     if ( !check.isRequired() )
429                     {
430                         item.result.addException( check.getException() );
431                         continue;
432                     }
433                 }
434 
435                 download.setChecksumPolicy( policy.getChecksumPolicy() );
436                 download.setRepositories( item.repository.getMirroredRepositories() );
437                 downloads.add( download );
438                 item.download = download;
439             }
440 
441             if ( downloads.isEmpty() )
442             {
443                 continue;
444             }
445 
446             for ( ArtifactDownload download : downloads )
447             {
448                 artifactDownloading( session, download.getTrace(), download.getArtifact(), group.repository );
449             }
450 
451             try
452             {
453                 RepositoryConnector connector =
454                     remoteRepositoryManager.getRepositoryConnector( session, group.repository );
455                 try
456                 {
457                     connector.get( downloads, null );
458                 }
459                 finally
460                 {
461                     connector.close();
462                 }
463             }
464             catch ( NoRepositoryConnectorException e )
465             {
466                 for ( ArtifactDownload download : downloads )
467                 {
468                     download.setException( new ArtifactTransferException( download.getArtifact(), group.repository, e ) );
469                 }
470             }
471 
472             for ( ResolutionItem item : group.items )
473             {
474                 ArtifactDownload download = item.download;
475                 if ( download == null )
476                 {
477                     continue;
478                 }
479 
480                 if ( item.updateCheck != null )
481                 {
482                     item.updateCheck.setException( download.getException() );
483                     updateCheckManager.touchArtifact( session, item.updateCheck );
484                 }
485 
486                 if ( download.getException() == null )
487                 {
488                     item.resolved.set( true );
489                     item.result.setRepository( group.repository );
490                     Artifact artifact = download.getArtifact();
491                     try
492                     {
493                         artifact = artifact.setFile( getFile( session, artifact, download.getFile() ) );
494                         item.result.setArtifact( artifact );
495                     }
496                     catch ( ArtifactTransferException e )
497                     {
498                         item.result.addException( e );
499                         continue;
500                     }
501                     lrm.add( session,
502                              new LocalArtifactRegistration( artifact, group.repository, download.getSupportedContexts() ) );
503 
504                     artifactDownloaded( session, download.getTrace(), artifact, group.repository, null );
505 
506                     artifactResolved( session, download.getTrace(), artifact, group.repository, null );
507                 }
508                 else
509                 {
510                     item.result.addException( download.getException() );
511 
512                     artifactDownloaded( session, download.getTrace(), download.getArtifact(), group.repository,
513                                         download.getException() );
514                 }
515             }
516         }
517 
518         for ( ArtifactResult result : results )
519         {
520             ArtifactRequest request = result.getRequest();
521 
522             Artifact artifact = result.getArtifact();
523             if ( artifact == null || artifact.getFile() == null )
524             {
525                 failures = true;
526                 if ( result.getExceptions().isEmpty() )
527                 {
528                     Exception exception = new ArtifactNotFoundException( request.getArtifact(), null );
529                     result.addException( exception );
530                 }
531                 RequestTrace trace = DefaultRequestTrace.newChild( request.getTrace(), request );
532                 artifactResolved( session, trace, request.getArtifact(), null, result.getExceptions() );
533             }
534         }
535 
536         if ( failures )
537         {
538             throw new ArtifactResolutionException( results );
539         }
540 
541         return results;
542     }

544     private boolean isLocallyInstalled( LocalArtifactResult lar, VersionResult vr )
545     {
546         if ( lar.isAvailable() )
547         {
548             return true;
549         }
550         if ( lar.getFile() != null )
551         {
552             if ( vr.getRepository() instanceof LocalRepository )
553             {
554                 // resolution of (snapshot) version found locally installed artifact
555                 return true;
556             }
557             else if ( vr.getRepository() == null && lar.getRequest().getRepositories().isEmpty() )
558             {
559                 // resolution of version range found locally installed artifact
560                 return true;
561             }
562         }
563         return false;
564     }

566     private File getFile( RepositorySystemSession session, Artifact artifact, File file )
567         throws ArtifactTransferException
568     {
569         if ( artifact.isSnapshot() && !artifact.getVersion().equals( artifact.getBaseVersion() )
570             && ConfigUtils.getBoolean( session, true, "aether.artifactResolver.snapshotNormalization" ) )
571         {
572             String name = file.getName().replace( artifact.getVersion(), artifact.getBaseVersion() );
573             File dst = new File( file.getParent(), name );
574 
575             boolean copy = dst.length() != file.length() || dst.lastModified() != file.lastModified();
576             if ( copy )
577             {
578                 try
579                 {
580                     fileProcessor.copy( file, dst, null );
581                     dst.setLastModified( file.lastModified() );
582                 }
583                 catch ( IOException e )
584                 {
585                     throw new ArtifactTransferException( artifact, null, e );
586                 }
587             }
588 
589             file = dst;
590         }
591 
592         return file;
593     }

643     static class ResolutionGroup
644     {
645 
646         final RemoteRepository repository;
647 
648         final List<ResolutionItem> items = new ArrayList<ResolutionItem>();
649 
650         ResolutionGroup( RemoteRepository repository )
651         {
652             this.repository = repository;
653         }
654 
655         boolean matches( RemoteRepository repo )
656         {
657             return repository.getUrl().equals( repo.getUrl() )
658                 && repository.getContentType().equals( repo.getContentType() )
659                 && repository.isRepositoryManager() == repo.isRepositoryManager();
660         }
661 
662     }

665     {
666 
667         final RequestTrace trace;
668 
669         final ArtifactRequest request;
670

672 
673         final LocalArtifactResult local;

677         final Artifact artifact;

685         ResolutionItem( RequestTrace trace, Artifact artifact, AtomicBoolean resolved, ArtifactResult result,
686                         LocalArtifactResult local, RemoteRepository repository )
687         {
688             this.trace = trace;
689             this.artifact = artifact;
690             this.resolved = resolved;
691             this.result = result;
692             this.request = result.getRequest();
693             this.local = local;
