199     public DeployResult deploy( RepositorySystemSession session, DeployRequest request )
200         throws DeploymentException
201     {
202         if ( session.isOffline() )
203         {
204             throw new DeploymentException( "The repository system is in offline mode, deployment impossible" );
205         }
206 
207         SyncContext syncContext = syncContextFactory.newInstance( session, false );
208 
209         try
210         {
211             return deploy( syncContext, session, request );
212         }
213         finally
214         {
215             syncContext.release();
216         }
217     }

219     private DeployResult deploy( SyncContext syncContext, RepositorySystemSession session, DeployRequest request )
220         throws DeploymentException
221     {
222         DeployResult result = new DeployResult( request );
223 
224         RequestTrace trace = DefaultRequestTrace.newChild( request.getTrace(), request );
225 
226         RemoteRepository repository = request.getRepository();
227 
228         RepositoryConnector connector;
229         try
230         {
231             connector = remoteRepositoryManager.getRepositoryConnector( session, repository );
232         }
233         catch ( NoRepositoryConnectorException e )
234         {
235             throw new DeploymentException( "Failed to deploy artifacts/metadata: " + e.getMessage(), e );
236         }
237 
238         List<MetadataGenerator> generators = getMetadataGenerators( session, request );
239 
240         try
241         {
242             List<ArtifactUpload> artifactUploads = new ArrayList<ArtifactUpload>();
243             List<MetadataUpload> metadataUploads = new ArrayList<MetadataUpload>();
244             IdentityHashMap<Metadata, Object> processedMetadata = new IdentityHashMap<Metadata, Object>();
245 
246             EventCatapult catapult = new EventCatapult( session, trace, repository, repositoryEventDispatcher );
247 
248             List<Artifact> artifacts = new ArrayList<Artifact>( request.getArtifacts() );
249 
250             List<Metadata> metadatas = Utils.prepareMetadata( generators, artifacts );
251 
252             syncContext.acquire( artifacts, Utils.combine( request.getMetadata(), metadatas ) );
253 
254             for ( Metadata metadata : metadatas )
255             {
256                 upload( metadataUploads, session, metadata, repository, connector, catapult );
257                 processedMetadata.put( metadata, null );
258             }
259 
260             for ( int i = 0; i < artifacts.size(); i++ )
261             {
262                 Artifact artifact = artifacts.get( i );
263 
264                 for ( MetadataGenerator generator : generators )
265                 {
266                     artifact = generator.transformArtifact( artifact );
267                 }
268 
269                 artifacts.set( i, artifact );
270 
271                 artifactUploads.add( new ArtifactUploadEx( artifact, artifact.getFile(), catapult ) );
272             }
273 
274             connector.put( artifactUploads, null );
275 
276             for ( ArtifactUpload upload : artifactUploads )
277             {
278                 if ( upload.getException() != null )
279                 {
280                     throw new DeploymentException( "Failed to deploy artifacts: " + upload.getException().getMessage(),
281                                                    upload.getException() );
282                 }
283                 result.addArtifact( upload.getArtifact() );
284             }
285 
286             metadatas = Utils.finishMetadata( generators, artifacts );
287 
288             syncContext.acquire( null, metadatas );
289 
290             for ( Metadata metadata : metadatas )
291             {
292                 upload( metadataUploads, session, metadata, repository, connector, catapult );
293                 processedMetadata.put( metadata, null );
294             }
295 
296             for ( Metadata metadata : request.getMetadata() )
297             {
298                 if ( !processedMetadata.containsKey( metadata ) )
299                 {
300                     upload( metadataUploads, session, metadata, repository, connector, catapult );
301                     processedMetadata.put( metadata, null );
302                 }
303             }
304 
305             connector.put( null, metadataUploads );
306 
307             for ( MetadataUpload upload : metadataUploads )
308             {
309                 if ( upload.getException() != null )
310                 {
311                     throw new DeploymentException( "Failed to deploy metadata: " + upload.getException().getMessage(),
312                                                    upload.getException() );
313                 }
314                 result.addMetadata( upload.getMetadata() );
315             }
316         }
317         finally
318         {
319             connector.close();
320         }
321 
322         return result;
323     }

325     private List<MetadataGenerator> getMetadataGenerators( RepositorySystemSession session, DeployRequest request )
326     {
327         List<MetadataGeneratorFactory> factories = Utils.sortMetadataGeneratorFactories( this.metadataFactories );
328 
329         List<MetadataGenerator> generators = new ArrayList<MetadataGenerator>();
330 
331         for ( MetadataGeneratorFactory factory : factories )
332         {
333             MetadataGenerator generator = factory.newInstance( session, request );
334             if ( generator != null )
335             {
336                 generators.add( generator );
337             }
338         }
339 
340         return generators;
341     }

343     private void upload( Collection<MetadataUpload> metadataUploads, RepositorySystemSession session,
344                          Metadata metadata, RemoteRepository repository, RepositoryConnector connector,
345                          EventCatapult catapult )
346         throws DeploymentException
347     {
348         LocalRepositoryManager lrm = session.getLocalRepositoryManager();
349         File basedir = lrm.getRepository().getBasedir();
350 
351         File dstFile = new File( basedir, lrm.getPathForRemoteMetadata( metadata, repository, "" ) );
352 
353         if ( metadata instanceof MergeableMetadata )
354         {
355             if ( !( (MergeableMetadata) metadata ).isMerged() )
356             {
357                 {
358                     DefaultRepositoryEvent event =
359                         new DefaultRepositoryEvent( EventType.METADATA_RESOLVING, session, catapult.getTrace() );
360                     event.setMetadata( metadata );
361                     event.setRepository( repository );
362                     repositoryEventDispatcher.dispatch( event );
363 
364                     event = new DefaultRepositoryEvent( EventType.METADATA_DOWNLOADING, session, catapult.getTrace() );
365                     event.setMetadata( metadata );
366                     event.setRepository( repository );
367                     repositoryEventDispatcher.dispatch( event );
368                 }
369 
370                 RepositoryPolicy policy = getPolicy( session, repository, metadata.getNature() );
371                 MetadataDownload download = new MetadataDownload();
372                 download.setMetadata( metadata );
373                 download.setFile( dstFile );
374                 download.setChecksumPolicy( policy.getChecksumPolicy() );
375                 connector.get( null, Arrays.asList( download ) );
376 
377                 Exception error = download.getException();
378 
379                 if ( error instanceof MetadataNotFoundException )
380                 {
381                     dstFile.delete();
382                 }
383 
384                 {
385                     DefaultRepositoryEvent event =
386                         new DefaultRepositoryEvent( EventType.METADATA_DOWNLOADED, session, catapult.getTrace() );
387                     event.setMetadata( metadata );
388                     event.setRepository( repository );
389                     event.setException( error );
390                     event.setFile( dstFile );
391                     repositoryEventDispatcher.dispatch( event );
392 
393                     event = new DefaultRepositoryEvent( EventType.METADATA_RESOLVED, session, catapult.getTrace() );
394                     event.setMetadata( metadata );
395                     event.setRepository( repository );
396                     event.setException( error );
397                     event.setFile( dstFile );
398                     repositoryEventDispatcher.dispatch( event );
399                 }
400 
401                 if ( error != null && !( error instanceof MetadataNotFoundException ) )
402                 {
403                     throw new DeploymentException( "Failed to retrieve remote metadata " + metadata + ": "
404                         + error.getMessage(), error );
405                 }
406             }
407 
408             try
409             {
410                 ( (MergeableMetadata) metadata ).merge( dstFile, dstFile );
411             }
412             catch ( RepositoryException e )
413             {
414                 throw new DeploymentException( "Failed to update metadata " + metadata + ": " + e.getMessage(), e );
415             }
416         }
417         else
418         {
419             if ( metadata.getFile() == null )
420             {
421                 throw new DeploymentException( "Failed to update metadata " + metadata + ": No file attached." );
422             }
423             try
424             {
425                 fileProcessor.copy( metadata.getFile(), dstFile, null );
426             }
427             catch ( IOException e )
428             {
429                 throw new DeploymentException( "Failed to update metadata " + metadata + ": " + e.getMessage(), e );
430             }
431         }
432 
433         UpdateCheck<Metadata, MetadataTransferException> check = new UpdateCheck<Metadata, MetadataTransferException>();
434         check.setItem( metadata );
435         check.setFile( dstFile );
436         check.setRepository( repository );
437         check.setAuthoritativeRepository( repository );
438         updateCheckManager.touchMetadata( session, check );
439 
440         metadataUploads.add( new MetadataUploadEx( metadata, dstFile, catapult ) );
441     }

443     private RepositoryPolicy getPolicy( RepositorySystemSession session, RemoteRepository repository,
444                                         Metadata.Nature nature )
445     {
446         boolean releases = !Metadata.Nature.SNAPSHOT.equals( nature );
447         boolean snapshots = !Metadata.Nature.RELEASE.equals( nature );
448         return remoteRepositoryManager.getPolicy( session, repository, releases, snapshots );
449     }

532         @Override
533         public Transfer setState( State state )
534         {
535             super.setState( state );
536 
537             if ( State.ACTIVE.equals( state ) )
538             {
539                 catapult.artifactDeploying( getArtifact(), getFile() );
540             }
541             else if ( State.DONE.equals( state ) )
542             {
543                 catapult.artifactDeployed( getArtifact(), getFile(), getException() );
544             }
545 
546             return this;
547         }
