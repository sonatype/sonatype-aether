50      @Test
51      public void testCopy()
52          throws IOException
53      {
54          File file = TestFileUtils.createTempFile( "testCopy\nasdf" );
55          File target = new File( targetDir, "testCopy.txt" );
56  
57          fileProcessor.copy( file, target, null );

61          file.delete();
62      }

64      @Test
65      public void testOverwrite()
66          throws IOException
67      {
68          File file = TestFileUtils.createTempFile( "testCopy\nasdf" );
69  
70          for ( int i = 0; i < 5; i++ )
71          {
72              File target = new File( targetDir, "testCopy.txt" );
73              fileProcessor.copy( file, target, null );
74              TestFileUtils.assertContent( "testCopy\nasdf".getBytes( "UTF-8" ), file );
75          }
76

78      }
79  
80      @Test
81      public void testCopyEmptyFile()
82          throws IOException
83      {
84          File file = TestFileUtils.createTempFile( "" );
85          File target = new File( targetDir, "testCopyEmptyFile" );
86          target.delete();
87          fileProcessor.copy( file, target, null );
88          assertTrue( "empty file was not copied", target.exists() && target.length() == 0 );
89          target.delete();

92      @Test
93      public void testProgressingChannel()
94          throws IOException
95      {
96          File file = TestFileUtils.createTempFile( "test" );
97          File target = new File( targetDir, "testProgressingChannel" );
98          target.delete();
99          final AtomicInteger progressed = new AtomicInteger();
100         ProgressListener listener = new ProgressListener()
101         {
102             public void progressed( ByteBuffer buffer )
103                 throws IOException
104             {
105                 progressed.addAndGet( buffer.remaining() );
106             }
107         };
108         fileProcessor.copy( file, target, listener );
109         assertTrue( "file was not created", target.isFile() );
110         assertEquals( "file was not fully copied", 4, target.length() );
111         assertEquals( "listener not called", 4, progressed.intValue() );
112         target.delete();
113     }
