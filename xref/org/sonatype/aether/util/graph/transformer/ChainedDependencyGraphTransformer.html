11  import org.sonatype.aether.RepositoryException;
12  import org.sonatype.aether.collection.DependencyGraphTransformationContext;
13  import org.sonatype.aether.collection.DependencyGraphTransformer;
14  import org.sonatype.aether.graph.DependencyNode;
15  
16  /**
17   * A dependency graph transformer that chains other transformers.
18   * 
19   * @author Benjamin Bentmann
20   */
21  public class ChainedDependencyGraphTransformer
22      implements DependencyGraphTransformer
23  {
24  
25      private final DependencyGraphTransformer[] transformers;
26  
27      /**
28       * Creates a new transformer that chains the specified transformers.
29       * 
30       * @param transformers The transformers to chain, may be {@code null} or empty.
31       */
32      public ChainedDependencyGraphTransformer( DependencyGraphTransformer... transformers )
33      {
34          if ( transformers == null )
35          {
36              this.transformers = new DependencyGraphTransformer[0];
37          }
38          else
39          {
40              this.transformers = transformers;
41          }
42      }
43  
44      /**
45       * Creates a new transformer that chains the specified transformers or simply returns one of them if the other one
46       * is {@code null}.
47       * 
48       * @param transformer1 The first transformer of the chain, may be {@code null}.
49       * @param transformer2 The second transformer of the chain, may be {@code null}.
50       * @return The chained transformer or {@code null} if both input transformers are {@code null}.
51       */
52      public static DependencyGraphTransformer newInstance( DependencyGraphTransformer transformer1,
53                                                            DependencyGraphTransformer transformer2 )
54      {
55          if ( transformer1 == null )
56          {
57              return transformer2;
58          }
59          else if ( transformer2 == null )
60          {
61              return transformer1;
62          }
63          return new ChainedDependencyGraphTransformer( transformer1, transformer2 );
64      }
65  
66      public DependencyNode transformGraph( DependencyNode node, DependencyGraphTransformationContext context )
67          throws RepositoryException
68      {
69          for ( DependencyGraphTransformer transformer : transformers )
70          {
71              node = transformer.transformGraph( node, context );
72          }
73          return node;
74      }
