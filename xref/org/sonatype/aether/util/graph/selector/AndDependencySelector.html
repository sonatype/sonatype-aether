20  /**
21   * A dependency selector that combines zero or more other selectors using a logical {@code AND}.
22   * 
23   * @author Benjamin Bentmann
24   */
25  public class AndDependencySelector
26      implements DependencySelector
27  {
28  
29      private final Collection<DependencySelector> selectors;
30  
31      /**
32       * Creates a new selector from the specified selectors.
33       * 
34       * @param selectors The selectors to combine, may be {@code null}.
35       */
36      public AndDependencySelector( DependencySelector... selectors )
37      {
38          if ( selectors != null && selectors.length > 0 )
39          {
40              this.selectors = new LinkedHashSet<DependencySelector>();
41              Collections.addAll( this.selectors, selectors );
42          }
43          else
44          {
45              this.selectors = Collections.emptySet();
46          }
47      }
48  
49      /**
50       * Creates a new selector from the specified selectors.
51       * 
52       * @param selectors The selectors to combine, may be {@code null}.
53       */
54      public AndDependencySelector( Set<DependencySelector> selectors )
55      {
56          if ( selectors != null && !selectors.isEmpty() )
57          {
58              this.selectors = selectors;
59          }
60          else
61          {
62              this.selectors = Collections.emptySet();
63          }
64      }
65  
66      /**
67       * Creates a new selector from the specified selectors.
68       * 
69       * @param selector1 The first selector to combine, may be {@code null}.
70       * @param selector2 The first selector to combine, may be {@code null}.
71       * @return The combined selector or {@code null} if both selectors were {@code null}.
72       */
73      public static DependencySelector newInstance( DependencySelector selector1, DependencySelector selector2 )
74      {
75          if ( selector1 == null )
76          {
77              return selector2;
78          }
79          else if ( selector2 == null )
80          {
81              return selector1;
82          }
83          return new AndDependencySelector( selector1, selector2 );
84      }
85  
86      public boolean selectDependency( Dependency dependency )
87      {
88          for ( DependencySelector selector : selectors )
89          {
90              if ( !selector.selectDependency( dependency ) )
91              {
92                  return false;
93              }
94          }
95          return true;
96      }
97  
98      public DependencySelector deriveChildSelector( DependencyCollectionContext context )
99      {
100         int seen = 0;
101         Set<DependencySelector> childSelectors = null;
102 
103         for ( DependencySelector selector : selectors )
104         {
105             DependencySelector childSelector = selector.deriveChildSelector( context );
106             if ( childSelectors != null )
107             {
108                 childSelectors.add( childSelector );
109             }
110             else if ( !selector.equals( childSelector ) )
111             {
112                 childSelectors = new LinkedHashSet<DependencySelector>();
113                 if ( seen > 0 )
114                 {
115                     for ( DependencySelector s : selectors )
116                     {
117                         if ( childSelectors.size() >= seen )
118                         {
119                             break;
120                         }
121                         childSelectors.add( s );
122                     }
123                 }
124                 childSelectors.add( childSelector );
125             }
126             else
127             {
128                 seen++;
129             }
130         }
131 
132         return childSelectors != null ? new AndDependencySelector( childSelectors ) : this;
133     }

144             return false;

148         return selectors.equals( that.selectors );
149     }
