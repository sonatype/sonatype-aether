30  /**
31   * Creates a dependency tree from a text description. <h2>Definition</h2> The description format is based on 'mvn
32   * dependency:tree'. Line format:
33   * 
34   * <pre>
35   * [level]dependencyDefinition[;key=value;key=value;...]
36   * </pre>
37   * 
38   * A <code>dependencyDefinition</code> is of the form:
39   * 
40   * <pre>
41   * (id|[(id)]gid:aid:ext:ver[:scope])
42   * </pre>
43   * 
44   * It may also be <code>(null)</code> to indicate an "empty" node with no dependency.
45   * <p>
46   * <h2>Levels</h2>
47   * <p>
48   * If <code>level</code> is empty, the line defines the root node. Only one root node may be defined. The level is
49   * calculated by the distance from the beginning of the line. One level is three characters. A level definition has to
50   * follow this format:
51   * 
52   * <pre>
53   * '[| ]*[+\\]- '
54   * </pre>
55   * 
56   * <h2>ID</h2> An ID may be used to reference a previously built node. An ID is of the form:
57   * 
58   * <pre>
59   * '[0-9a-zA-Z]+'
60   * </pre>
61   * 
62   * To define a node with an ID, prefix the definition with an id in parens:
63   * 
64   * <pre>
65   * (id)gid:aid:ext:ver
66   * </pre>
67   * 
68   * To insert a previously defined node into the graph, use a caret followed by the ID:
69   * 
70   * <pre>
71   * ^id
72   * </pre>
73   * 
74   * <h2>Comments</h2>
75   * <p>
76   * A hash starts a comment. A comment ends with the end of the line. Empty lines are ignored.
77   * <h2>Example</h2>
78   * 
79   * <pre>
80   * gid:aid:ext:ver
81   * +- gid:aid2:ext:ver:scope
82   * |  \- (id1)gid:aid3:ext:ver
83   * +- gid:aid4:ext:ver:scope
84   * \- ^id1
85   * </pre>
86   * 
87   * <h2>Multiple definitions in one resource</h2>
88   * <p>
89   * By using {@link #parseMultiple(String)}, definitions divided by a line beginning with "---" can be read from the same
90   * resource. The rest of the line is ignored.
91   * <h2>Substitutions</h2>
92   * <p>
93   * You may define substitutions (see {@link #setSubstitutions(String...)},
94   * {@link #DependencyGraphParser(String, Collection)}). Every '%s' in the definition will be substituted by the next
95   * String in the defined substitutions.
96   * <h3>Example</h3>
97   * 
98   * <pre>
99   * parser.setSubstitutions( &quot;foo&quot;, &quot;bar&quot; );
100  * String def = &quot;gid:%s:ext:ver\n&quot; + &quot;+- gid:%s:ext:ver&quot;;
101  * </pre>
102  * 
103  * The first node will have "foo" as its artifact id, the second node (child to the first) will have "bar" as its
104  * artifact id.
105  * 
106  * @author Benjamin Hanzelmann
107  */

218     private DependencyNode parse( BufferedReader in )
219         throws IOException
220     {
221 
222         if ( substitutions != null )
223         {
224             substitutionIterator = substitutions.iterator();
225         }
226 
227         String line = null;
228 
229         DependencyNode root = null;
230         DependencyNode node = null;
231         int prevLevel = 0;
232 
233         LinkedList<DependencyNode> stack = new LinkedList<DependencyNode>();
234         boolean isRootNode = true;
235 
236         while ( ( line = in.readLine() ) != null )
237         {
238             line = cutComment( line );
239 
240             if ( isEmpty( line ) )
241             {
242                 // skip empty line
243                 continue;
244             }
245 
246             if ( isEOFMarker( line ) )
247             {
248                 // stop parsing
249                 break;
250             }
251 
252             while ( line.contains( "%s" ) )
253             {
254                 if ( !substitutionIterator.hasNext() )
255                 {
256                     throw new IllegalArgumentException( "not enough substitutions to fill placeholders" );
257                 }
258                 line = line.replaceFirst( "%s", substitutionIterator.next() );
259             }
260 
261             LineContext ctx = createContext( line );
262             if ( prevLevel < ctx.getLevel() )
263             {
264                 // previous node is new parent
265                 stack.add( node );
266             }
267 
268             // get to real parent
269             while ( prevLevel > ctx.getLevel() )
270             {
271                 stack.removeLast();
272                 prevLevel -= 1;
273             }
274 
275             prevLevel = ctx.getLevel();
276 
277             if ( ctx.getDefinition() != null && ctx.getDefinition().isReference() )
278             {
279                 DependencyNode child = reference( ctx.getDefinition().getReference() );
280                 node.getChildren().add( child );
281                 node = child;
282             }
283             else
284             {
285 
286                 node = build( isRootNode ? null : stack.getLast(), ctx, isRootNode );
287 
288                 if ( isRootNode )
289                 {
290                     root = node;
291                     isRootNode = false;
292                 }
293 
294                 if ( ctx.getDefinition() != null && ctx.getDefinition().hasId() )
295                 {
296                     this.nodes.put( ctx.getDefinition().getId(), node );
297                 }
298             }
299         }
300 
301         this.nodes.clear();
302 
303         return root;
304 
305     }

339     private DependencyNode build( DependencyNode parent, LineContext ctx, boolean isRoot )
340     {
341         ArtifactDefinition def = ctx.getDefinition();
342         if ( !isRoot && parent == null )
343         {
344             throw new IllegalArgumentException( "dangling node: " + def );
345         }
346         else if ( ctx.getLevel() == 0 && parent != null )
347         {
348             throw new IllegalArgumentException( "inconsistent leveling (parent for level 0?): " + def );
349         }
350 
351         NodeBuilder builder = new NodeBuilder();
352 
353         if ( def != null )
354         {
355             builder.artifactId( def.getArtifactId() ).groupId( def.getGroupId() );
356             builder.ext( def.getExtension() ).version( def.getVersion() ).scope( def.getScope() );
357             builder.properties( ctx.getProperties() );
358         }
359         DependencyNode node = builder.build();
360 
361         if ( parent != null )
362         {
363             parent.getChildren().add( node );
364         }
365 
366         return node;
367     }

369     public String dump( DependencyNode root )
370     {
371         StringBuilder ret = new StringBuilder();
372 
373         List<NodeEntry> entries = new ArrayList<NodeEntry>();
374 
375         addNode( root, 0, entries );
376 
377         for ( NodeEntry nodeEntry : entries )
378         {
379             char[] level = new char[( nodeEntry.getLevel() * 3 )];
380             Arrays.fill( level, ' ' );
381 
382             if ( level.length != 0 )
383             {
384                 level[level.length - 3] = '+';
385                 level[level.length - 2] = '-';
386             }
387 
388             String definition = nodeEntry.getDefinition();
389 
390             ret.append( level ).append( definition ).append( "\n" );
391         }
392 
393         return ret.toString();
394 
395     }

397     private void addNode( DependencyNode root, int level, List<NodeEntry> entries )
398     {
399 
400         NodeEntry entry = new NodeEntry();
401         Dependency dependency = root.getDependency();
402         StringBuilder defBuilder = new StringBuilder();
403         if ( dependency == null )
404         {
405             defBuilder.append( "(null)" );
406         }
407         else
408         {
409             Artifact artifact = dependency.getArtifact();
410 
411             defBuilder.append( artifact.getGroupId() ).append( ":" ).append( artifact.getArtifactId() ).append( ":" ).append( artifact.getExtension() ).append( ":" ).append( artifact.getVersion() );
412             if ( dependency.getScope() != null && ( !"".equals( dependency.getScope() ) ) )
413             {
414                 defBuilder.append( ":" ).append( dependency.getScope() );
415             }
416 
417             Map<String, String> properties = artifact.getProperties();
418             if ( !( properties == null || properties.isEmpty() ) )
419             {
420                 for ( Map.Entry<String, String> prop : properties.entrySet() )
421                 {
422                     defBuilder.append( ";" ).append( prop.getKey() ).append( "=" ).append( prop.getValue() );
423                 }
424             }
425         }
426 
427         entry.setDefinition( defBuilder.toString() );
428         entry.setLevel( level++ );
429 
430         entries.add( entry );
431 
432         for ( DependencyNode node : root.getChildren() )
433         {
434             addNode( node, level, entries );
435         }

478     private static LineContext createContext( String line )
479     {
480         LineContext ctx = new LineContext();
481         String definition;
482 
483         String[] split = line.split( "- " );
484         if ( split.length == 1 ) // root
485         {
486             ctx.setLevel( 0 );
487             definition = split[0];
488         }
489         else
490         {
491             ctx.setLevel( (int) Math.ceil( (double) split[0].length() / (double) 3 ) );
492             definition = split[1];
493         }
494 
495         if ( "(null)".equalsIgnoreCase( definition ) )
496         {
497             return ctx;
498         }
499 
500         split = definition.split( ";" );
501         ctx.setDefinition( new ArtifactDefinition( split[0] ) );
502 
503         if ( split.length > 1 ) // properties
504         {
505             Map<String, String> props = new HashMap<String, String>();
506             for ( int i = 1; i < split.length; i++ )
507             {
508                 String[] keyValue = split[i].split( "=" );
509                 String key = keyValue[0];
510                 String value = keyValue[1];
511                 props.put( key, value );
512             }
513             ctx.setProperties( props );
514         }
515 
516         return ctx;
517     }
