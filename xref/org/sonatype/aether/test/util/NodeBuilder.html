120     public DependencyNode build()
121     {
122         Dependency dependency = null;
123         TestDependencyNode node = new TestDependencyNode();
124         if ( artifactId != null && artifactId.length() > 0 )
125         {
126             Artifact artifact = new StubArtifact( groupId, artifactId, classifier, ext, version, properties );
127             dependency = new Dependency( artifact, scope, optional );
128             node.setDependency( dependency );
129             try
130             {
131                 node.setVersion( versionScheme.parseVersion( version ) );
132                 node.setVersionConstraint( versionScheme.parseVersionConstraint( range != null ? range : version ) );
133             }
134             catch ( InvalidVersionSpecificationException e )
135             {
136                 throw new IllegalArgumentException( "bad version: " + e.getMessage(), e );
137             }
138         }
139         node.setRequestContext( context );
140         node.setRelocations( relocations );
141         return node;
142     }
