158     public List<MetadataResult> resolveMetadata( RepositorySystemSession session,
159                                                  Collection<? extends MetadataRequest> requests )
160     {
161         SyncContext syncContext = syncContextFactory.newInstance( session, false );
162 
163         try
164         {
165             Collection<Metadata> metadata = new ArrayList<Metadata>( requests.size() );
166             for ( MetadataRequest request : requests )
167             {
168                 metadata.add( request.getMetadata() );
169             }
170 
171             syncContext.acquire( null, metadata );
172 
173             return resolve( session, requests );
174         }
175         finally
176         {
177             syncContext.release();
178         }
179     }

181     private List<MetadataResult> resolve( RepositorySystemSession session,
182                                           Collection<? extends MetadataRequest> requests )
183     {
184         List<MetadataResult> results = new ArrayList<MetadataResult>( requests.size() );
185 
186         List<ResolveTask> tasks = new ArrayList<ResolveTask>( requests.size() );
187 
188         Map<File, Long> localLastUpdates = new HashMap<File, Long>();
189 
190         for ( MetadataRequest request : requests )
191         {
192             RequestTrace trace = DefaultRequestTrace.newChild( request.getTrace(), request );
193 
194             MetadataResult result = new MetadataResult( request );
195             results.add( result );
196 
197             Metadata metadata = request.getMetadata();
198             RemoteRepository repository = request.getRepository();
199 
200             if ( repository == null )
201             {
202                 LocalRepository localRepo = session.getLocalRepositoryManager().getRepository();
203 
204                 metadataResolving( session, trace, metadata, localRepo );
205 
206                 File localFile = getLocalFile( session, metadata );
207 
208                 if ( localFile != null )
209                 {
210                     metadata = metadata.setFile( localFile );
211                     result.setMetadata( metadata );
212                 }
213                 else
214                 {
215                     result.setException( new MetadataNotFoundException( metadata, localRepo ) );
216                 }
217 
218                 metadataResolved( session, trace, metadata, localRepo, result.getException() );
219                 continue;
220             }
221 
222             List<RemoteRepository> repositories = getEnabledSourceRepositories( repository, metadata.getNature() );
223 
224             if ( repositories.isEmpty() )
225             {
226                 continue;
227             }
228 
229             metadataResolving( session, trace, metadata, repository );
230             LocalRepositoryManager lrm = session.getLocalRepositoryManager();
231             LocalMetadataRequest localRequest =
232                 new LocalMetadataRequest( metadata, repository, request.getRequestContext() );
233             LocalMetadataResult lrmResult = lrm.find( session, localRequest );
234 
235             File metadataFile = lrmResult.getFile();
236 
237             if ( session.isOffline() )
238             {
239                 if ( metadataFile != null )
240                 {
241                     metadata = metadata.setFile( metadataFile );
242                     result.setMetadata( metadata );
243                 }
244                 else
245                 {
246                     String msg =
247                         "The repository system is offline but the metadata " + metadata + " from " + repository
248                             + " is not available in the local repository.";
249                     result.setException( new MetadataNotFoundException( metadata, repository, msg ) );
250                 }
251 
252                 metadataResolved( session, trace, metadata, repository, result.getException() );
253                 continue;
254             }
255 
256             Long localLastUpdate = null;
257             if ( request.isFavorLocalRepository() )
258             {
259                 File localFile = getLocalFile( session, metadata );
260                 localLastUpdate = localLastUpdates.get( localFile );
261                 if ( localLastUpdate == null )
262                 {
263                     localLastUpdate = Long.valueOf( localFile != null ? localFile.lastModified() : 0 );
264                     localLastUpdates.put( localFile, localLastUpdate );
265                 }
266             }
267 
268             List<UpdateCheck<Metadata, MetadataTransferException>> checks =
269                 new ArrayList<UpdateCheck<Metadata, MetadataTransferException>>();
270             Exception exception = null;
271             for ( RemoteRepository repo : repositories )
272             {
273                 UpdateCheck<Metadata, MetadataTransferException> check =
274                     new UpdateCheck<Metadata, MetadataTransferException>();
275                 check.setLocalLastUpdated( ( localLastUpdate != null ) ? localLastUpdate.longValue() : 0 );
276                 check.setItem( metadata );
277 
278                 // use 'main' installation file for the check (-> use requested repository)
279                 File checkFile =
280                     new File(
281                               session.getLocalRepository().getBasedir(),
282                               session.getLocalRepositoryManager().getPathForRemoteMetadata( metadata, repository,
283                                                                                             request.getRequestContext() ) );
284                 check.setFile( checkFile );
285                 check.setRepository( repository );
286                 check.setAuthoritativeRepository( repo );
287                 check.setPolicy( getPolicy( session, repo, metadata.getNature() ).getUpdatePolicy() );
288 
289                 if ( lrmResult.isStale() )
290                 {
291                     checks.add( check );
292                 }
293                 else
294                 {
295                     updateCheckManager.checkMetadata( session, check );
296                     if ( check.isRequired() )
297                     {
298                         checks.add( check );
299                     }
300                     else if ( exception == null )
301                     {
302                         exception = check.getException();
303                     }
304                 }
305             }
306 
307             if ( !checks.isEmpty() )
308             {
309                 RepositoryPolicy policy = getPolicy( session, repository, metadata.getNature() );
310 
311                 // install path may be different from lookup path
312                 File installFile =
313                     new File(
314                               session.getLocalRepository().getBasedir(),
315                               session.getLocalRepositoryManager().getPathForRemoteMetadata( metadata,
316                                                                                             request.getRepository(),
317                                                                                             request.getRequestContext() ) );
318 
319                 ResolveTask task =
320                     new ResolveTask( session, trace, result, installFile, checks, policy.getChecksumPolicy() );
321                 tasks.add( task );
322             }
323             else
324             {
325                 result.setException( exception );
326                 if ( metadataFile != null )
327                 {
328                     metadata = metadata.setFile( metadataFile );
329                     result.setMetadata( metadata );
330                 }
331                 metadataResolved( session, trace, metadata, repository, result.getException() );
332             }
333         }
334 
335         if ( !tasks.isEmpty() )
336         {
337             int threads = ConfigUtils.getInteger( session, 4, "aether.metadataResolver.threads" );
338             Executor executor = getExecutor( Math.min( tasks.size(), threads ) );
339             try
340             {
341                 RunnableErrorForwarder errorForwarder = new RunnableErrorForwarder();
342 
343                 for ( ResolveTask task : tasks )
344                 {
345                     executor.execute( errorForwarder.wrap( task ) );
346                 }
347 
348                 errorForwarder.await();
349 
350                 for ( ResolveTask task : tasks )
351                 {
352                     task.result.setException( task.exception );
353                 }
354             }
355             finally
356             {
357                 shutdown( executor );
358             }
359             for ( ResolveTask task : tasks )
360             {
361                 Metadata metadata = task.request.getMetadata();
362                 // re-lookup metadata for resolve
363                 LocalMetadataRequest localRequest =
364                     new LocalMetadataRequest( metadata, task.request.getRepository(), task.request.getRequestContext() );
365                 File metadataFile = session.getLocalRepositoryManager().find( session, localRequest ).getFile();
366                 if ( metadataFile != null )
367                 {
368                     metadata = metadata.setFile( metadataFile );
369                     task.result.setMetadata( metadata );
370                 }
371                 if ( task.result.getException() == null )
372                 {
373                     task.result.setUpdated( true );
374                 }
375                 metadataResolved( session, task.trace, metadata, task.request.getRepository(),
376                                   task.result.getException() );
377             }
378         }
379 
380         return results;
381     }

391     private List<RemoteRepository> getEnabledSourceRepositories( RemoteRepository repository, Metadata.Nature nature )
392     {
393         List<RemoteRepository> repositories = new ArrayList<RemoteRepository>();
394 
395         if ( repository.isRepositoryManager() )
396         {
397             for ( RemoteRepository repo : repository.getMirroredRepositories() )
398             {
399                 if ( isEnabled( repo, nature ) )
400                 {
401                     repositories.add( repo );
402                 }
403             }
404         }
405         else if ( isEnabled( repository, nature ) )
406         {
407             repositories.add( repository );
408         }
409 
410         return repositories;
411     }

413     private boolean isEnabled( RemoteRepository repository, Metadata.Nature nature )
414     {
415         if ( !Metadata.Nature.SNAPSHOT.equals( nature ) && repository.getPolicy( false ).isEnabled() )
416         {
417             return true;
418         }
419         if ( !Metadata.Nature.RELEASE.equals( nature ) && repository.getPolicy( true ).isEnabled() )
420         {
421             return true;
422         }
423         return false;
424     }

426     private RepositoryPolicy getPolicy( RepositorySystemSession session, RemoteRepository repository,
427                                         Metadata.Nature nature )
428     {
429         boolean releases = !Metadata.Nature.SNAPSHOT.equals( nature );
430         boolean snapshots = !Metadata.Nature.RELEASE.equals( nature );
431         return remoteRepositoryManager.getPolicy( session, repository, releases, snapshots );
432     }

478     private Executor getExecutor( int threads )
479     {
480         if ( threads <= 1 )
481         {
482             return new Executor()
483             {
484                 public void execute( Runnable command )
485                 {
486                     command.run();
487                 }
488             };
489         }
490         else
491         {
492             return new ThreadPoolExecutor( threads, threads, 3, TimeUnit.SECONDS, new LinkedBlockingQueue<Runnable>() );
493         }
494     }

504     class ResolveTask
505         implements Runnable
506     {
507 
508         final RepositorySystemSession session;
509 
510         final RequestTrace trace;
511 
512         final MetadataResult result;
513 
514         final MetadataRequest request;
515 
516         final File metadataFile;
517 
518         final String policy;
519 
520         final List<UpdateCheck<Metadata, MetadataTransferException>> checks;
521 
522         volatile MetadataTransferException exception;
523 
524         public ResolveTask( RepositorySystemSession session, RequestTrace trace, MetadataResult result,
525                             File metadataFile, List<UpdateCheck<Metadata, MetadataTransferException>> checks,
526                             String policy )
527         {
528             this.session = session;
529             this.trace = trace;
530             this.result = result;
531             this.request = result.getRequest();
532             this.metadataFile = metadataFile;
533             this.policy = policy;
534             this.checks = checks;
535         }
536 
537         public void run()
538         {
539             Metadata metadata = request.getMetadata();
540             RemoteRepository requestRepository = request.getRepository();
541 
542             metadataDownloading( session, trace, metadata, requestRepository );
543 
544             try
545             {
546                 List<RemoteRepository> repositories = new ArrayList<RemoteRepository>();
547                 for ( UpdateCheck<Metadata, MetadataTransferException> check : checks )
548                 {
549                     repositories.add( check.getAuthoritativeRepository() );
550                 }
551 
552                 MetadataDownload download = new MetadataDownload();
553                 download.setMetadata( metadata );
554                 download.setRequestContext( request.getRequestContext() );
555                 download.setFile( metadataFile );
556                 download.setChecksumPolicy( policy );
557                 download.setRepositories( repositories );
558 
559                 RepositoryConnector connector =
560                     remoteRepositoryManager.getRepositoryConnector( session, requestRepository );
561                 try
562                 {
563                     connector.get( null, Arrays.asList( download ) );
564                 }
565                 finally
566                 {
567                     connector.close();
568                 }
569 
570                 exception = download.getException();
571 
572                 if ( exception == null )
573                 {
574 
575                     List<String> contexts = Collections.singletonList( request.getRequestContext() );
576                     LocalMetadataRegistration registration =
577                         new LocalMetadataRegistration( metadata, requestRepository, contexts );
578 
579                     session.getLocalRepositoryManager().add( session, registration );
580                 }
581                 else if ( request.isDeleteLocalCopyIfMissing() && exception instanceof MetadataNotFoundException )
582                 {
583                     download.getFile().delete();
584                 }
585             }
586             catch ( NoRepositoryConnectorException e )
587             {
588                 exception = new MetadataTransferException( metadata, requestRepository, e );
589             }
590 
591             for ( UpdateCheck<Metadata, MetadataTransferException> check : checks )
592             {
593                 updateCheckManager.touchMetadata( session, check.setException( exception ) );
594             }
595 
596             metadataDownloaded( session, trace, metadata, requestRepository, metadataFile, exception );
597         }
598 
599     }
